### Vertical Alignment: default = "middle"
If no `options`/`options.verticalAlign` prop is defined, the paths will be vertically centered.

<Preview>
    <Story name="verticalAlign: middle">
        <FlowModeler
            flow={{
                elements: {
                    "1": { data: { label: "First Node" }, nextElementId: "2" },
                    "2": { data: { label: "Second Node" }, nextElementId: "3" },
                    "3": { data: { label: "Condition" }, nextElements: [
                        { id: "4.1", conditionData: { label: "Fulfilled" } },
                        { id: "4.2", conditionData: { label: "Not fulfilled" } }] },
                    "4.1": { data: { label: "Optional Node 1" }, nextElementId: "5.1" },
                    "4.2": { data: { label: "Optional Node 2" }},
                    "5.1": { data: { label: "Last Node"}}
                },
                firstElementId: "1"
            }}
            renderContent={(data, id) => <span>{data.label}</span>}
            renderGatewayConditionType={(data, id) => <label>{data.label}</label>}
            renderGatewayConditionValue={(data, id) => <label>{data.label}</label>}
        />
    </Story>
</Preview>
