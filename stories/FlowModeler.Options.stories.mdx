    <Story name="verticalAlign: top">
        <FlowModeler
            flow={{
                firstElementId: "1",
                elements: {
                    "1": { data: { label: "First" }, nextElementId: "2" },
                    "2": { data: { label: "Second" }, nextElementId: "3" },
                    "3": { data: { label: "Condition" }, nextElements: [
                        { id: "4.1", conditionData: { label: "Fulfilled" } },
                        { id: "4.2", conditionData: { label: "Not fulfilled" } }] },
                    "4.1": { data: { label: "Option 1" }, nextElementId: "5.1" },
                    "4.2": { data: { label: "Option 2" }},
                    "5.1": { data: { label: "Follow-Up"}}
                }
            }}
            options={{ verticalAlign: "top" }}
            renderContent={({ elementData }) => <span>{elementData.label}</span>}
            renderGatewayConditionType={({ gatewayData }) => <label>{gatewayData.label}</label>}
            renderGatewayConditionValue={({ conditionData }) => <label>{conditionData.label}</label>}
        />
    </Story>
</Preview>
